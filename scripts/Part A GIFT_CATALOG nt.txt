CREATE OR REPLACE TYPE ty_gift_item AS OBJECT (
    name VARCHAR2(100)
);
/

CREATE OR REPLACE TYPE gift_box AS TABLE OF ty_gift_item;
/

CREATE TABLE gift_catalog (
    gift_id      NUMBER        PRIMARY KEY,
    min_purchase NUMBER        NOT NULL,
    gifts        gift_box
)
NESTED TABLE gifts STORE AS nt_gift_storage;
/
